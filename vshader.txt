#version 330

//Zmienne jednorodne
uniform mat4 P;
uniform mat4 V;
uniform mat4 M;


//Atrybuty
in vec3 vertex; //wspolrzedne wierzcholka w przestrzeni modelu
in vec3 normal; //wektor normalny w wierzcholku (na razie ignorowany)
in vec2 texCoords0; //współrzędne teksturowania

vec4 lp=vec4(0,0,7,1); //Położenie źródła światła w przestrzeni świata

out vec4 iV;
out vec4 iN;
out vec4 iL;
out vec2 iTexCoords0;

void main(void) {
	gl_Position=P*V*M*vec4(vertex.xyz,1);
	iN=normalize(V*M*vec4(normal.xyz,1)); //wektor normalny w przestrzeni oka
	iV=normalize(vec4(0,0,0,1)-V*M*vec4(vertex.xyz,1)); //wektor do obserwatora w przestrzeni oka
	iL=normalize(V*lp-V*M*vec4(vertex.xyz,1)); //wektor do światła w przestrzeni oka
	iTexCoords0=texCoords0;
}
